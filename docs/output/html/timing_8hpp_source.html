56// Simple wall/CPU timer
57class Timer {
58public:
59 using clock = std::chrono::steady_clock;
60
61 void start() {
62 if (running_) return;
63 running_ = true;
64 t0_ = clock::now();
65 cpu0_ = std::clock();
66 }
67
68 // Stop and return last interval (seconds)
69 double stop() {
70 if (!running_) return 0.0;
71 const auto t1 = clock::now();
72 const double dt = std::chrono::duration<double>(t1 - t0_).count();
73 accumulated_ += dt;
74 running_ = false;
75
76 const std::clock_t cpu1 = std::clock();
77 if (cpu1 != (std::clock_t)-1 && cpu0_ != (std::clock_t)-1) {
78 accumulated_cpu_ += double(cpu1 - cpu0_) / double(CLOCKS_PER_SEC);
79 }
80 return dt;
81 }
82
83 void reset() { running_ = false; accumulated_ = 0.0; accumulated_cpu_ = 0.0; }
84
85 double elapsed() const {
86 if (!running_) return accumulated_;
87 return accumulated_ + std::chrono::duration<double>(clock::now() - t0_).count();
88 }
89
90 double cpu_elapsed() const { return accumulated_cpu_; }
91
92private:
93 clock::time_point t0_{};
94 std::clock_t cpu0_{};
95 double accumulated_ {0.0};
96 double accumulated_cpu_{0.0};
97 bool running_ {false};
98};

167// Convenience: time a callable (averaged over `repeat` runs)
168template<class F>
169inline double time_it(F&& fn, int warmup=1, int repeat=3) {
170 for (int i = 0; i < warmup; ++i) fn();
171 Timer t; t.start();
172 for (int i = 0; i < repeat; ++i) fn();
173 t.stop();
174 return t.elapsed() / double(repeat);
175}
